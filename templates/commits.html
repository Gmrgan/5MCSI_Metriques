async function fetchCommits() { try { const response = await fetch("/api/commits/"); const data = await response.json(); if (data.error) { console.error("Erreur API :", data.error); return; } const labels = Object.keys(data).sort(); // Trier par heure:minute const values = labels.map(minute => data[minute]); // Remplir le tableau HTML const tableBody = document.getElementById("commitsTableBody"); labels.forEach(minute => { const row = `${minute}${data[minute]}`; tableBody.innerHTML += row; }); // Afficher le graphique const ctx = document.getElementById("commitsChart").getContext("2d"); new Chart(ctx, { type: "bar", data: { labels: labels, datasets: [{ label: "Nombre de commits", data: values, backgroundColor: "rgba(54, 162, 235, 0.6)", borderColor: "rgba(54, 162, 235, 1)", borderWidth: 1 }] }, options: { responsive: true, scales: { y: { beginAtZero: true } } } }); } catch (error) { console.error("Erreur de chargement des commits :", error); } } fetchCommits();
